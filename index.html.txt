function spawnUnits(type) {
  if (!state.running) return;
  const def = UNIT_DEFS[type];
  if (!def) return;
  if (state.resource < def.cost) { addLog('군량이 부족합니다!', 'event'); return; }
  const now = Date.now();
  if (state.cooldowns[type] && now - state.cooldowns[type] < COOLDOWN) return;

  state.resource -= def.cost;
  state.cooldowns[type] = now;
  document.getElementById('resourceVal').textContent = state.resource;

  const count = 10;
  const sp = def.faction === 'defender' ? state.defenderSpawn : state.attackerSpawn;

  for (let i = 0; i < count; i++) {
    const ang = Math.random() * Math.PI * 2;

    state.units.push({
      id: state.nextId++,
      type,
      ...def,
      hp: def.hp,
      maxHp: def.hp,
      x: sp.x + Math.cos(ang) * r,

      vx: 0, vy: 0,
      target: null,
      targetZone: null,
      attackTimer: 0,
      alive: true,
    });
  }
  addLog(`${def.name} 10명 소환`, def.faction);
  startCooldownUI(type);
}

function startCooldownUI(type) {
  const btn = document.getElementById(`btn-${type}`);
  if (!btn) return;
  const end = Date.now() + COOLDOWN;
  const interval = setInterval(() => {
    const left = Math.ceil((end - Date.now()) / 1000);
    if (left <= 0) {
      btn.querySelector('.cooldown-overlay')?.remove();
      clearInterval(interval);
      return;
    }
    let overlay = btn.querySelector('.cooldown-overlay');
    if (!overlay) {
      overlay = document.createElement('div');
      overlay.className = 'cooldown-overlay';
      btn.appendChild(overlay);
    }
    overlay.textContent = left + 's';
  }, 100);
}

function updateAI(unit, dt) {
  if (!unit.alive) return;

  const enemies = state.units.filter(u => u.alive && u.faction !== unit.faction);

  // Find closest enemy
  let closest = null, closestDist = Infinity;
  for (const e of enemies) {
    const d = dist(unit, e);
    if (d < closestDist) { closestDist = d; closest = e; }
  }

  // Pick target zone
  let targetZone = null;
  if (unit.faction === 'attacker') {
    const uncaptured = state.zones.filter(z => z.owner !== 'attacker');
    targetZone = uncaptured.length ? uncaptured[0] : state.zones[0];
  } else {
    const threatened = state.zones.filter(z => z.owner !== 'defender');
    targetZone = threatened.length ? threatened[0] : state.zones[Math.floor(Math.random() * state.zones.length)];
  }

  const zoneRange = targetZone ? targetZone.radius + 10 : 999;
  const zd = targetZone ? dist(unit, targetZone) : 999;

  let mx = 0, my = 0;

  if (unit.role === 'ranged') {
    // ── RANGED AI ──
    const preferredDist = unit.range * 0.7; // stay at ~70% of max range
    const minDist = 35; // don't let melee enemies get too close

    if (closest) {
      if (closestDist < minDist) {
        // Kite backwards
        const dx = unit.x - closest.x, dy = unit.y - closest.y;
        const d = Math.hypot(dx, dy) || 1;
        mx = dx / d; my = dy / d;
      } else if (closestDist > unit.range) {
        // Move closer to shoot
        const dx = closest.x - unit.x, dy = closest.y - unit.y;
        const d = Math.hypot(dx, dy) || 1;
        mx = dx / d * 0.5; my = dy / d * 0.5;
      }
      // else stay put and shoot
    } else if (targetZone && zd > zoneRange + 40) {
      const dx = targetZone.x - unit.x, dy = targetZone.y - unit.y;
      const d = Math.hypot(dx, dy) || 1;
      mx = dx / d * 0.6; my = dy / d * 0.6;
    }
  } else {
    // ── MELEE AI ──
    if (closest && closestDist < unit.range * 2.5) {
      const dx = closest.x - unit.x, dy = closest.y - unit.y;
      const d = Math.hypot(dx, dy) || 1;
      mx = dx / d; my = dy / d;
    } else if (targetZone && zd > zoneRange) {
      const dx = targetZone.x - unit.x, dy = targetZone.y - unit.y;
      const d = Math.hypot(dx, dy) || 1;
      mx = dx / d; my = dy / d;
    }
  }

  unit.vx += mx * 0.6;
  unit.vy += my * 0.6;
  unit.vx *= 0.82;
  unit.vy *= 0.82;

  const maxSpd = unit.spd * 1.4;
  const spd = Math.hypot(unit.vx, unit.vy);
  if (spd > maxSpd) { unit.vx *= maxSpd/spd; unit.vy *= maxSpd/spd; }

  unit.x += unit.vx;
  unit.y += unit.vy;
  unit.x = Math.max(unit.size, Math.min(W - unit.size, unit.x));
  unit.y = Math.max(unit.size, Math.min(H - 150, unit.y));

  // ── ATTACK ──
  unit.attackTimer -= dt;
  const cd = unit.atkCooldown || (unit.role === 'ranged' ? 1.8 : 1.0);

  if (closest && closestDist < unit.range && unit.attackTimer <= 0) {
    unit.attackTimer = cd * (0.85 + Math.random() * 0.3);

    if (unit.role === 'ranged') {
      // Fire projectile
      const dx = closest.x - unit.x, dy = closest.y - unit.y;
      const d = Math.hypot(dx, dy) || 1;
      state.projectiles.push({
        x: unit.x, y: unit.y,
        vx: (dx / d) * (unit.projSpd || 6),
        vy: (dy / d) * (unit.projSpd || 6),
        target: closest,
        atk: unit.atk,
        color: unit.projColor || unit.color,
        size: unit.projSize || 2.5,
        life: 2.0,
        faction: unit.faction,
      });
    } else {
      // Melee hit
      closest.hp -= unit.atk * (0.8 + Math.random() * 0.4);
      spawnParticle(closest.x, closest.y, closest.color);
      if (closest.hp <= 0) {
        closest.alive = false;
        spawnDeathParticles(closest.x, closest.y, closest.color);
      }
    }
  }
}

// ─────────────────────────────────────────
//  PROJECTILE UPDATE
// ─────────────────────────────────────────
function updateProjectiles(dt) {
  for (const p of state.projectiles) {
    p.x += p.vx;
    p.y += p.vy;
    p.life -= dt;

    // Check hit against enemies
    for (const u of state.units) {
      if (!u.alive || u.faction === p.faction) continue;
      if (dist(p, u) < u.size + p.size) {
        u.hp -= p.atk * (0.8 + Math.random() * 0.4);
        spawnParticle(u.x, u.y, p.color);
        if (u.hp <= 0) {
          u.alive = false;
          spawnDeathParticles(u.x, u.y, u.color);
        }
        p.life = 0; // consume projectile
        break;
      }
    }
  }
  state.projectiles = state.projectiles.filter(p => p.life > 0 && p.x > 0 && p.x < W && p.y > 0 && p.y < H);
}

// ─────────────────────────────────────────
//  ZONE CAPTURE LOGIC
// ─────────────────────────────────────────
function updateZones(dt) {
  for (const zone of state.zones) {
    zone.defCount = 0; zone.atkCount = 0;
    for (const u of state.units) {
      if (!u.alive) continue;
      if (dist(u, zone) < zone.radius) {
        if (u.faction === 'attacker') zone.atkCount++;
        else zone.defCount++;
      }
    }

    const prevOwner = zone.owner;

    if (zone.atkCount > 0 && zone.defCount === 0) {
      zone.progress = Math.min(1, zone.progress + dt * 0.08);
      if (zone.progress >= 1) zone.owner = 'attacker';
      else zone.owner = 'contested';
    } else if (zone.defCount > 0 && zone.atkCount === 0) {
      zone.progress = Math.max(0, zone.progress - dt * 0.08);
      if (zone.progress <= 0) zone.owner = 'defender';
      else zone.owner = 'contested';
    } else if (zone.atkCount > 0 && zone.defCount > 0) {
      zone.owner = 'contested';
    }

    // Score
    if (zone.owner === 'attacker') state.atkScore += dt * 2;
    else if (zone.owner === 'defender') state.defScore += dt * 2;

    if (prevOwner !== zone.owner && (zone.owner === 'attacker' || zone.owner === 'defender')) {
      addLog(`${zone.name} ${zone.owner==='attacker'?'일본군 점령!':'조선·명군 탈환!'}`, zone.owner==='attacker'?'attacker':'defender');
    }

    // Update UI
    const fill = document.getElementById(`zfill-${zone.id}`);
    const status = document.getElementById(`zstatus-${zone.id}`);
    if (fill) {
      fill.style.width = (zone.progress * 100) + '%';
      fill.style.background = zone.owner==='attacker' ? '#c0392b' : zone.owner==='contested' ? '#c9923a' : '#2d6a4f';
    }
    if (status) {
      status.className = `zone-status ${zone.owner==='defender'?'held':zone.owner==='contested'?'contested':'captured'}`;
      status.textContent = zone.owner==='defender'?'수비중':zone.owner==='contested'?'교전중':'점령됨';
    }
  }
}

  // Ground
  ctx.fillStyle = '#1a1400';
  ctx.fillRect(0, 0, W, H);

  // Terrain texture
  ctx.save();
  ctx.globalAlpha = 0.07;
  for (let i = 0; i < 200; i++) {
    ctx.fillStyle = Math.random() > 0.5 ? '#8b7040' : '#4a3000';
    ctx.fillRect(Math.random()*W, Math.random()*H, 2+Math.random()*6, 2+Math.random()*6);
  }
  ctx.restore();

  // Castle outer ground
  ctx.save();
  ctx.fillStyle = '#2a1e08';
  ctx.fillRect(cx - w/2 - thick, cy - h/2 - thick, w + thick*2, h + thick*2);
  ctx.restore();

  // Wall shadow
  ctx.save();
  ctx.shadowColor = '#000';
  ctx.shadowBlur = 24;
  ctx.fillStyle = '#3a2808';
  ctx.fillRect(cx - w/2 - thick, cy - h/2 - thick, w + thick*2, h + thick*2);
  ctx.restore();

  // Castle interior (courtyard)
  ctx.fillStyle = '#221800';
  ctx.fillRect(cx - w/2, cy - h/2, w, h);

  // Interior detail
  ctx.save();
  ctx.strokeStyle = '#3a2a10';
  ctx.lineWidth = 1;
  ctx.setLineDash([6, 8]);
  for (let iy = cy - h/2 + 30; iy < cy + h/2; iy += 40) {
    ctx.beginPath(); ctx.moveTo(cx - w/2, iy); ctx.lineTo(cx + w/2, iy); ctx.stroke();
  }
  ctx.setLineDash([]);
  ctx.restore();

  // Wall top (battlements hint)
  ctx.save();
  ctx.strokeStyle = '#6a5020';
  ctx.lineWidth = thick;
  ctx.strokeRect(cx - w/2, cy - h/2, w, h);
  ctx.strokeStyle = '#8b6830';
  ctx.lineWidth = 2;
  ctx.strokeRect(cx - w/2, cy - h/2, w, h);

  // Battlements dots
  ctx.fillStyle = '#5a4018';
  const bStep = 28;
  for (let bx = cx - w/2; bx < cx + w/2; bx += bStep) {
    ctx.fillRect(bx, cy - h/2 - thick/2 - 6, 14, 8);
    ctx.fillRect(bx, cy + h/2 + thick/2 - 2, 14, 8);
  }
  for (let by = cy - h/2; by < cy + h/2; by += bStep) {
    ctx.fillRect(cx - w/2 - thick/2 - 6, by, 8, 14);
    ctx.fillRect(cx + w/2 + thick/2 - 2, by, 8, 14);
  }
  ctx.restore();

  // Gate (south)
  ctx.fillStyle = '#1a0d00';
  ctx.fillRect(cx - 20, cy + h/2 - thick/2 - 2, 40, thick + 4);
  ctx.fillStyle = '#8b6830';
  ctx.fillRect(cx - 18, cy + h/2 - thick/2 - 1, 36, 3);

  // Zones
  for (const z of state.zones) {
    ctx.save();
    const alpha = 0.18 + 0.06 * Math.sin(Date.now() / 800);
    ctx.globalAlpha = alpha;
    ctx.fillStyle = z.owner === 'attacker' ? '#c0392b' : z.owner === 'contested' ? '#c9923a' : '#2d6a4f';
    ctx.beginPath();
    ctx.arc(z.x, z.y, z.radius, 0, Math.PI*2);
    ctx.fill();
    ctx.globalAlpha = 0.7;
    ctx.strokeStyle = z.owner === 'attacker' ? '#ff4444' : z.owner === 'contested' ? '#ffd700' : '#40c070';
    ctx.lineWidth = 1.5;
    ctx.setLineDash([6, 4]);
    ctx.stroke();
    ctx.setLineDash([]);
    ctx.restore();

    // Zone label
    ctx.save();
    ctx.font = 'bold 11px "Noto Serif KR"';
    ctx.fillStyle = '#f0d080';
    ctx.textAlign = 'center';
    ctx.shadowColor = '#000';
    ctx.shadowBlur = 4;
    ctx.fillText(z.name, z.x, z.y + 4);
    ctx.restore();

    // Progress arc
    if (z.progress > 0 && z.progress < 1) {
      ctx.save();
      ctx.strokeStyle = '#c0392b';
      ctx.lineWidth = 4;
      ctx.beginPath();
      ctx.arc(z.x, z.y, z.radius - 4, -Math.PI/2, -Math.PI/2 + z.progress * Math.PI * 2);
      ctx.stroke();
      ctx.restore();
    }
  }

  // Spawn points
  drawSpawnPoint(state.defenderSpawn, '#40c070', '수비 집결지');
  drawSpawnPoint(state.attackerSpawn, '#e04030', '공격 집결지');
}

function drawSpawnPoint(sp, color, label) {
  ctx.save();
  ctx.strokeStyle = color;
  ctx.lineWidth = 1.5;
  ctx.globalAlpha = 0.5 + 0.2 * Math.sin(Date.now() / 600);
  ctx.setLineDash([4, 6]);
  ctx.beginPath();
  ctx.arc(sp.x, sp.y, 36, 0, Math.PI*2);
  ctx.stroke();
  ctx.setLineDash([]);
  ctx.fillStyle = color;
  ctx.globalAlpha = 0.1;
  ctx.beginPath();
  ctx.arc(sp.x, sp.y, 36, 0, Math.PI*2);
  ctx.fill();
  ctx.globalAlpha = 0.7;
  ctx.font = '10px "Noto Sans KR"';
  ctx.fillStyle = color;
  ctx.textAlign = 'center';
  ctx.fillText(label, sp.x, sp.y - 42);
  ctx.restore();
}

function drawProjectiles() {
  for (const p of state.projectiles) {
    ctx.save();
    ctx.beginPath();
    ctx.arc(p.x, p.y, p.size, 0, Math.PI * 2);
    ctx.fillStyle = p.color;
    ctx.shadowColor = p.color;
    ctx.shadowBlur = 10;
    ctx.fill();
    // Trail
    ctx.globalAlpha = 0.3;
    ctx.beginPath();
    ctx.arc(p.x - p.vx * 1.5, p.y - p.vy * 1.5, p.size * 0.7, 0, Math.PI * 2);
    ctx.fill();
    ctx.restore();
  }
}

function drawUnits() {
  for (const u of state.units) {
    if (!u.alive) continue;
    ctx.save();
    // Shadow
    ctx.beginPath();
    ctx.ellipse(u.x, u.y + u.size, u.size * 0.9, u.size * 0.4, 0, 0, Math.PI * 2);
    ctx.fillStyle = 'rgba(0,0,0,0.4)';
    ctx.fill();

    // Body — diamond shape for ranged, circle for melee
    if (u.role === 'ranged') {
      ctx.beginPath();
      ctx.moveTo(u.x, u.y - u.size * 1.3);
      ctx.lineTo(u.x + u.size * 1.1, u.y);
      ctx.lineTo(u.x, u.y + u.size * 1.3);
      ctx.lineTo(u.x - u.size * 1.1, u.y);
      ctx.closePath();
      ctx.fillStyle = u.color;
      ctx.shadowColor = u.color;
      ctx.shadowBlur = 10;
      ctx.fill();
      // Inner dot
      ctx.beginPath();
      ctx.arc(u.x, u.y, u.size * 0.4, 0, Math.PI * 2);
      ctx.fillStyle = 'rgba(255,255,255,0.5)';
      ctx.shadowBlur = 0;
      ctx.fill();
    } else {
      ctx.beginPath();
      ctx.arc(u.x, u.y, u.size, 0, Math.PI * 2);
      ctx.fillStyle = u.color;
      ctx.shadowColor = u.color;
      ctx.shadowBlur = 8;
      ctx.fill();
    }

    // HP bar
    const barW = u.size * 2.8;
    const barH = 3;
    const bx = u.x - barW / 2;
    const by = u.y - u.size * 1.5 - 5;
    ctx.shadowBlur = 0;
    ctx.fillStyle = '#1a0a0a';
    ctx.fillRect(bx, by, barW, barH);
    const ratio = Math.max(0, u.hp / u.maxHp);
    ctx.fillStyle = ratio > 0.6 ? '#40c070' : ratio > 0.3 ? '#c9923a' : '#c0392b';
    ctx.fillRect(bx, by, barW * ratio, barH);
    ctx.restore();
  }
}

function drawParticles() {
  for (const p of state.particles) {
    const alpha = p.life / p.maxLife;
    ctx.save();
    ctx.globalAlpha = alpha * 0.9;
    ctx.fillStyle = p.color;
    ctx.beginPath();
    ctx.arc(p.x, p.y, p.size * alpha, 0, Math.PI * 2);
    ctx.fill();
    ctx.restore();
  }
}

function drawMinimap() {
  const sc = 160 / W, sc2 = 110 / H;
  mmCtx.fillStyle = '#0a0600';
  mmCtx.fillRect(0, 0, 160, 110);

  // Castle rect
  const { cx, cy, w, h, thick } = state.castle;
  mmCtx.fillStyle = '#3a2808';
  mmCtx.fillRect((cx-w/2-thick)*sc, (cy-h/2-thick)*sc2, (w+thick*2)*sc, (h+thick*2)*sc2);
  mmCtx.fillStyle = '#1a1000';
  mmCtx.fillRect((cx-w/2)*sc, (cy-h/2)*sc2, w*sc, h*sc2);

  // Zones
  for (const z of state.zones) {
    mmCtx.fillStyle = z.owner==='attacker'?'#c0392b':z.owner==='contested'?'#c9923a':'#2d6a4f';
    mmCtx.beginPath();
    mmCtx.arc(z.x*sc, z.y*sc2, 6, 0, Math.PI*2);
    mmCtx.fill();
  }

  // Units
  for (const u of state.units) {
    if (!u.alive) continue;
    mmCtx.fillStyle = u.color;
    mmCtx.fillRect(u.x*sc - 1, u.y*sc2 - 1, 2, 2);
  }
}

// ─────────────────────────────────────────
//  TIMER
// ─────────────────────────────────────────
function updateTimer(dt) {
  state.time -= dt;
  if (state.time <= 0) { state.time = 0; endGame(); return; }
  const m = Math.floor(state.time / 60);
  const s = Math.floor(state.time % 60);
  const td = document.getElementById('timerDisplay');
  td.textContent = `${String(m).padStart(2,'0')}:${String(s).padStart(2,'0')}`;
  td.className = state.time <= 120 ? 'urgent' : '';
}

function updateScoreUI() {
  document.getElementById('atkScore').textContent = Math.round(state.atkScore);
  document.getElementById('defScore').textContent = Math.round(state.defScore);
}

// ─────────────────────────────────────────
//  LOG
// ─────────────────────────────────────────
function addLog(text, type = '') {
  const panel = document.getElementById('logPanel');
  const entry = document.createElement('div');
  entry.className = `log-entry ${type}`;
  entry.textContent = text;
  panel.prepend(entry);
  state.logs.push(entry);
  if (state.logs.length > 8) {
    panel.removeChild(state.logs.shift());
  }
}

// ─────────────────────────────────────────
//  MAIN LOOP
// ─────────────────────────────────────────
let lastTime = 0;
function loop(ts) {
  if (!state.running) return;
  const dt = Math.min((ts - lastTime) / 1000, 0.1);
  lastTime = ts;

  // Update units
  for (const u of state.units) updateAI(u, dt);
  state.units = state.units.filter(u => u.alive);

  updateProjectiles(dt);

  // Particles
  for (const p of state.particles) {
    p.x += p.vx; p.y += p.vy;
    p.life -= dt;
  }
  state.particles = state.particles.filter(p => p.life > 0);

  updateZones(dt);
  autoSpawn(dt);
  regenResource(dt);
  updateTimer(dt);
  updateScoreUI();

  // Draw
  ctx.clearRect(0, 0, W, H);
  drawMap();
  drawUnits();
  drawProjectiles();
  drawParticles();
  drawMinimap();

  requestAnimationFrame(loop);
}

// ─────────────────────────────────────────
//  GAME START / END
// ─────────────────────────────────────────
function startGame() {
  document.getElementById('overlay').style.display = 'none';
  state.running = true;
  state.time = GAME_DURATION;
  state.resource = 200;
  state.units = [];
  state.particles = [];
  state.projectiles = [];
  state.atkScore = 0;
  state.defScore = 0;
  state.cooldowns = {};
  autoSpawnTimer = 5;
  buildMap();

  // Initial forces
  setTimeout(() => spawnInitialForces(), 500);

  lastTime = performance.now();
  requestAnimationFrame(loop);
}

function spawnInitialForces() {
  const defTypes = ['joseon_militia','joseon_regular','joseon_archer_militia','ming_regular','ming_archer_militia'];
  const atkTypes = ['japan_ashigaru_low','japan_ashigaru_high','japan_gunner_militia'];
  for (const t of defTypes) spawnGroup(t, 10, state.defenderSpawn);
  for (const t of atkTypes) spawnGroup(t, 10, state.attackerSpawn);
}

function spawnGroup(type, count, sp) {
  const def = UNIT_DEFS[type];
  for (let i = 0; i < count; i++) {
    const ang = Math.random() * Math.PI * 2;
    const r = 20 + Math.random() * 40;
    state.units.push({
      id: state.nextId++,
      type,
      ...def,
      hp: def.hp,
      maxHp: def.hp,
      x: sp.x + Math.cos(ang) * r,
      y: sp.y + Math.sin(ang) * r,
      vx: 0, vy: 0,
      target: null,
      targetZone: null,
      attackTimer: 0,
      alive: true,
    });
  }
}

function endGame() {
  state.running = false;
  const capturedZones = state.zones.filter(z => z.owner === 'attacker').length;
  const totalZones = state.zones.length;
  const atkWin = capturedZones >= 2; // attackers need 2/3 zones

  const overlay = document.getElementById('overlay');
  overlay.style.display = 'flex';
  overlay.innerHTML = `
    <div class="overlay-title" style="color:${atkWin?'#c0392b':'#2d9e6f'}">
      ${atkWin ? '일본군 승리' : '조선·명군 승리'}
    </div>
    <div class="overlay-sub">
      ${atkWin
        ? `<span class="result-attacker">일본군이 성을 함락하였습니다!</span><br>점령구역: ${capturedZones}/${totalZones}`
        : `<span class="result-defender">조선·명 연합군이 성을 지켜냈습니다!</span><br>점령구역: ${capturedZones}/${totalZones}`
      }<br><br>
      공격 점수: <b style="color:#c0392b">${Math.round(state.atkScore)}</b>
      &nbsp;|&nbsp;
      수비 점수: <b style="color:#2d9e6f">${Math.round(state.defScore)}</b>
    </div>
    <button class="overlay-btn" onclick="startGame()">재전투 ▶</button>
  `;
}
